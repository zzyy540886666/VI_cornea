### 3.3 迁移学习 + 分层冻结

**实现代码**：

```python
import torch
import torch.nn as nn
from timm import create_model

class CornealClassifier(nn.Module):
    """角膜地形图分类器"""
    
    def __init__(self, model_name='convnextv2_base', pretrained=True, num_classes=2):
        super().__init__()
        # 加载 ImageNet 预训练模型
        self.backbone = create_model(
            model_name, 
            pretrained=pretrained,
            num_classes=num_classes,
            drop_rate=0.5,
            drop_path_rate=0.2
        )
    
    def forward(self, x):
        return self.backbone(x)

def create_model_with_freezing(stage='head'):
    """
    创建模型并设置冻结策略
    
    Args:
        stage: 训练阶段 ('head' 或 'fine_tune')
    """
    model = CornealClassifier(model_name='convnextv2_base', pretrained=True)
    
    if stage == 'head':
        # 第一阶段：只训练分类头
        for name, param in model.named_parameters():
            if 'head.fc' not in name:
                param.requires_grad = False
        print(f"已冻结特征提取器，只训练分类头")
        
    elif stage == 'fine_tune':
        # 第二阶段：解冻最后阶段进行微调
        for name, param in model.named_parameters():
            if 'head.fc' not in name and 'stages.3' not in name:
                param.requires_grad = False
        print(f"已解冻 stages.3，进行微调")
    
    return model

# 使用示例
# 阶段 1：训练分类头
model = create_model_with_freezing(stage='head')
optimizer = torch.optim.AdamW(
    filter(lambda p: p.requires_grad, model.parameters()),
    lr=1e-3,
    weight_decay=0.05
)
train(epochs=10)  # 训练 10 个 epoch

# 阶段 2：微调
model = create_model_with_freezing(stage='fine_tune')
optimizer = torch.optim.AdamW(
    filter(lambda p: p.requires_grad, model.parameters()),
    lr=1e-4,
    weight_decay=0.05
)
train(epochs=30)  # 微调 30 个 epoch
```

**训练策略**：
- **阶段 1** (10 epochs, lr=1e-3): 只训练分类头，快速适配新任务
- **阶段 2** (30 epochs, lr=1e-4): 微调最后阶段，提升性能

### 3.4 防过拟合策略

**实现代码**：

```python
import torch
import torch.nn as nn
from torch.optim import AdamW
from torch.optim.lr_scheduler import CosineAnnealingLR

class EarlyStopping:
    """早停法实现"""
    
    def __init__(self, patience=10, min_delta=0.001):
        self.patience = patience
        self.min_delta = min_delta
        self.counter = 0
        self.best_loss = None
        self.early_stop = False
    
    def __call__(self, val_loss):
        if self.best_loss is None:
            self.best_loss = val_loss
        elif val_loss > self.best_loss - self.min_delta:
            self.counter += 1
            if self.counter >= self.patience:
                self.early_stop = True
        else:
            self.best_loss = val_loss
            self.counter = 0

def get_optimizer_and_scheduler(model):
    """获取优化器和学习率调度器"""
    optimizer = AdamW(
        model.parameters(),
        lr=1e-4,
        weight_decay=0.05,  # L2 正则化
        betas=(0.9, 0.999)
    )
    
    scheduler = CosineAnnealingLR(
        optimizer,
        T_max=50,
        eta_min=1e-6
    )
    
    return optimizer, scheduler

# 训练循环示例
def train_one_epoch(model, train_loader, val_loader, device):
    model.train()
    total_loss = 0
    
    for images, labels in train_loader:
        images, labels = images.to(device), labels.to(device)
        
        optimizer.zero_grad()
        outputs = model(images)
        loss = criterion(outputs, labels)
        loss.backward()
        
        # 梯度裁剪（防止梯度爆炸）
        torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
        
        optimizer.step()
        total_loss += loss.item()
    
    return total_loss / len(train_loader)

# 主训练循环
early_stopping = EarlyStopping(patience=10)
best_accuracy = 0.0

for epoch in range(100):
    train_loss = train_one_epoch(model, train_loader, val_loader, device)
    val_accuracy = validate(model, val_loader, device)
    
    # 保存最佳模型
    if val_accuracy > best_accuracy:
        best_accuracy = val_accuracy
        torch.save(model.state_dict(), 'best_model.pth')
    
    # 早停检查
    early_stopping(train_loss)
    if early_stopping.early_stop:
        print(f"早停：验证集损失 {early_stopping.counter} 个 epoch 未提升")
        break
    
    scheduler.step()
```

**防过拟合技术**：
1. **Dropout**: drop_rate=0.5, drop_path_rate=0.2
2. **L2 正则化**: weight_decay=0.05
3. **梯度裁剪**: max_norm=1.0
4. **早停法**: patience=10
5. **学习率调度**: CosineAnnealingLR

### 3.5 5 折交叉验证

**实现代码**：

```python
from sklearn.model_selection import KFold
import numpy as np

def k_fold_cross_validation(dataset, k=5, random_state=42):
    """
    5 折交叉验证
    
    Args:
        dataset: 完整数据集
        k: 折数
        random_state: 随机种子
    
    Returns:
        scores: 每折的评估结果
    """
    kfold = KFold(n_splits=k, shuffle=True, random_state=random_state)
    scores = []
    
    for fold, (train_idx, test_idx) in enumerate(kfold.split(dataset)):
        print(f"\n{'='*50}")
        print(f"Fold {fold + 1}/{k}")
        print(f"{'='*50}")
        
        # 划分训练集和测试集
        train_dataset = torch.utils.data.Subset(dataset, train_idx)
        test_dataset = torch.utils.data.Subset(dataset, test_idx)
        
        # 创建数据加载器
        train_loader = torch.utils.data.DataLoader(
            train_dataset, 
            batch_size=16, 
            shuffle=True
        )
        test_loader = torch.utils.data.DataLoader(
            test_dataset, 
            batch_size=16, 
            shuffle=False
        )
        
        # 训练模型
        model = create_model_with_freezing(stage='head')
        train(model, train_loader, epochs=50)
        
        # 评估
        accuracy = evaluate(model, test_loader)
        scores.append(accuracy)
        
        print(f"Fold {fold + 1} Accuracy: {accuracy:.4f}")
    
    # 计算平均准确率和标准差
    mean_accuracy = np.mean(scores)
    std_accuracy = np.std(scores)
    
    print(f"\n{'='*50}")
    print(f"5 折交叉验证结果")
    print(f"{'='*50}")
    print(f"平均准确率：{mean_accuracy:.4f} ± {std_accuracy:.4f}")
    print(f"最高准确率：{max(scores):.4f}")
    print(f"最低准确率：{min(scores):.4f}")
    
    return scores

# 使用示例
scores = k_fold_cross_validation(full_dataset, k=5)
```

```python
import torch
from pytorch_grad_cam import GradCAMPlusPlus
from pytorch_grad_cam.utils import image_to_tensor, reshape_transform
import cv2
import numpy as np

class CornealGradCAM:
    """角膜地形图 Grad-CAM 解释器"""
    
    def __init__(self, model, device='cuda'):
        self.model = model.to(device)
        self.device = device
        # 选择最后一层卷积层作为目标层
        self.target_layers = [model.backbone.head.fc]
        self.cam = GradCAMPlusPlus(
            model=self.model,
            target_layers=self.target_layers,
            reshape_transform=reshape_transform
        )
    
    def generate_heatmap(self, image, pred_class=None):
        """
        生成热力图
        
        Args:
            image: 输入图像 (numpy array, HWC)
            pred_class: 预测类别，如果为 None 则使用模型预测
        
        Returns:
            heatmap: 热力图
        """
        # 预处理
        input_tensor = image_to_tensor(image).unsqueeze(0).to(self.device)
        
        # 如果未指定类别，使用模型预测
        if pred_class is None:
            with torch.no_grad():
                outputs = self.model(input_tensor)
                pred_class = outputs.argmax(dim=1).item()
        
        # 生成热力图
        grayscale_cam = self.cam(
            input_tensor=input_tensor,
            targets=[ClassifierOutputTarget(pred_class)]
        )[0]
        
        # 可视化
        visualization = show_cam_on_image(
            image.astype(np.float32) / 255.0,
            grayscale_cam,
            use_rgb=True
        )
        
        return grayscale_cam, visualization
    
    def analyze_key_regions(self, heatmap, threshold=0.5):
        """
        分析关键区域
        
        Args:
            heatmap: 热力图
            threshold: 注意力阈值
        
        Returns:
            regions: 关键区域列表
        """
        # 二值化
        binary = (heatmap > threshold).astype(np.uint8) * 255
        
        # 连通区域分析
        contours, _ = cv2.findContours(
            binary, 
            cv2.RETR_EXTERNAL, 
            cv2.CHAIN_APPROX_SIMPLE
        )
        
        regions = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if area > 100:  # 最小区域阈值
                x, y, w, h = cv2.boundingRect(contour)
                attention_score = heatmap[y:y+h, x:x+w].mean()
                
                regions.append({
                    'position': (x, y),
                    'size': (w, h),
                    'attention_score': attention_score,
                    'area': area
                })
        
        # 按注意力分数排序
        regions.sort(key=lambda x: x['attention_score'], reverse=True)
        
        return regions

# 使用示例
```

```python
def generate_decision_path(metrics):
    """
    生成决策路径
    
    Args:
        metrics: 临床指标字典
            - kmax: 最大曲率 (D)
            - is_value: I-S 值 (D)
            - cct: 角膜中央厚度 (μm)
    
    Returns:
        path: 决策路径列表
    """
    path = []
    abnormal_count = 0
    
    # Step 1: 检查 Kmax
    if metrics['kmax'] > 46.0:
        path.append(f"✓ Kmax: {metrics['kmax']}D > 46.0D → 异常")
        abnormal_count += 1
    else:
        path.append(f"✗ Kmax: {metrics['kmax']}D < 46.0D → 正常")
    
    # Step 2: 检查 I-S 值
    if metrics['is_value'] > 1.8:
        path.append(f"✓ I-S 值：{metrics['is_value']}D > 1.8D → 异常")
        abnormal_count += 1
    else:
        path.append(f"✗ I-S 值：{metrics['is_value']}D < 1.8D → 正常")
    
    # Step 3: 检查 CCT
    if metrics['cct'] < 500:
        path.append(f"✓ CCT: {metrics['cct']}μm < 500μm → 偏薄")
        abnormal_count += 1
    else:
        path.append(f"✗ CCT: {metrics['cct']}μm > 500μm → 正常")
    
    # 综合判断
    if abnormal_count >= 2:
        diagnosis = "异常（圆锥角膜）"
    elif abnormal_count == 1:
        diagnosis = "临界（建议复查）"
    else:
        diagnosis = "正常"
    
    path.append(f"\n综合判断：{diagnosis}")
    path.append(f"异常指标数：{abnormal_count}/3")
    
    return path

# 使用示例
metrics = {

    'cct': 485
}

    print(step)
```

### 4.3 多维度解释整合

**实现代码**：

```python
class ExplainablePredictor:
    """可解释预测器"""
    
    def __init__(self, model, device='cuda'):
        self.model = model.to(device)
        self.device = device
        self.grad_cam = CornealGradCAM(model, device)
    
    def predict_and_explain(self, image, metrics):
        """
        预测并提供解释
        
        Args:
            image: 角膜地形图
            metrics: 临床指标
        
        Returns:
            result: 包含预测和解释的字典
        """
        # 1. 模型预测
        self.model.eval()
        input_tensor = image_to_tensor(image).unsqueeze(0).to(self.device)
        
        with torch.no_grad():
            outputs = self.model(input_tensor)
            probs = torch.softmax(outputs, dim=1)[0]
            pred_class = outputs.argmax(dim=1).item()
            confidence = probs[pred_class].item()
        
        # 2. Grad-CAM 热力图
        heatmap, visualization = self.grad_cam.generate_heatmap(
            image, 
            pred_class
        )
        key_regions = self.grad_cam.analyze_key_regions(heatmap)
        
        # 3. 决策路径
        decision_path = generate_decision_path(metrics)
        
        # 4. 整合结果
        result = {
            'prediction': {
                'class': pred_class,
                'label': '异常（圆锥角膜）' if pred_class == 1 else '正常',
                'confidence': confidence,
                'probabilities': probs.cpu().numpy()
            },
            'explanations': {
                'heatmap': heatmap,
                'visualization': visualization,
                'key_regions': key_regions,
                'decision_path': decision_path,
                'metrics': metrics
            }
        }
        
        return result

# 使用示例
predictor = ExplainablePredictor(model)
result = predictor.predict_and_explain(image, metrics)

print(f"预测结果：{result['prediction']['label']}")
```

## 五、多模型集成

### 5.1 集成架构

```python
import torch
import torch.nn as nn
from timm import create_model

class EnsembleClassifier(nn.Module):
    """集成分类器"""
    
    def __init__(self, model_names=['convnextv2_base', 'maxvit_t', 'swinv2_base'], 
                 num_classes=2, device='cuda'):
        super().__init__()
        self.device = device
        self.model_names = model_names
        
        # 创建多个模型
        self.models = nn.ModuleList()
        for name in model_names:
            model = create_model(
                name,
                pretrained=True,
                num_classes=num_classes
            )
            self.models.append(model.to(device))
        
        # 模型权重（可学习）
        self.model_weights = nn.Parameter(
            torch.ones(len(model_names)) / len(model_names)
        )
    
    def forward(self, x):
        """加权投票"""
        all_outputs = []
        
        for model in self.models:
            outputs = model(x)
            all_outputs.append(outputs)
        
        # 加权平均
        stacked_outputs = torch.stack(all_outputs, dim=0)  # [num_models, batch, num_classes]
        weights = torch.softmax(self.model_weights, dim=0)  # 归一化权重
        weighted_outputs = torch.einsum('mbc,m->bc', stacked_outputs, weights)
        
        return weighted_outputs
    
    def predict_proba(self, x):
        """预测概率"""
        self.eval()
        with torch.no_grad():
            outputs = self.forward(x)
            probs = torch.softmax(outputs, dim=1)
        return probs.cpu().numpy()

# 使用示例
ensemble = EnsembleClassifier(
    model_names=['convnextv2_base', 'maxvit_t', 'swinv2_base'],
    num_classes=2
)

# 训练集成模型
criterion = nn.CrossEntropyLoss()
optimizer = torch.optim.AdamW(
    ensemble.parameters(),
    lr=1e-4,
    weight_decay=0.05
)

# 训练循环
for epoch in range(50):
    for images, labels in train_loader:
        images, labels = images.to(device), labels.to(device)
        
        optimizer.zero_grad()
        outputs = ensemble(images)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()
```

### 6.1 完整训练 pipeline

```python
import torch
from torch.utils.data import DataLoader
from torchvision import datasets, transforms

class TrainingPipeline:
    """训练流程"""
    
    def __init__(self, config):
        self.config = config
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        
    def prepare_data(self):
        """准备数据"""
        # 数据增强
        train_transform = get_train_transforms()
        val_transform = transforms.Compose([
            transforms.Resize(256),
            transforms.CenterCrop(224),
            transforms.ToTensor(),
            transforms.Normalize(mean=[0.485, 0.456, 0.406], 
                               std=[0.229, 0.224, 0.225])
        ])
        
        # 加载数据
        train_dataset = datasets.ImageFolder(
            self.config['data_root'],
            transform=train_transform
        )
        val_dataset = datasets.ImageFolder(
            self.config['data_root'],
            transform=val_transform
        )
        
        # 创建数据加载器
        train_loader = DataLoader(
            train_dataset,
            batch_size=self.config['batch_size'],
            shuffle=True,
            num_workers=4
        )
        val_loader = DataLoader(
            val_dataset,
            batch_size=self.config['batch_size'],
            shuffle=False,
            num_workers=4
        )
        
        return train_loader, val_loader
    
    def train(self):
        """完整训练流程"""
        # 1. 准备数据
        train_loader, val_loader = self.prepare_data()
        
        # 2. 创建模型
        model = create_model_with_freezing(stage='head')
        model = model.to(self.device)
        
        # 3. 定义损失函数和优化器
        criterion = nn.CrossEntropyLoss()
        optimizer, scheduler = get_optimizer_and_scheduler(model)
        
        # 4. 训练
        best_accuracy = 0.0
        for epoch in range(self.config['epochs']):
            # 训练一个 epoch
            train_loss = self.train_one_epoch(
                model, train_loader, optimizer, criterion
            )
            
            # 验证
            val_accuracy = self.validate(model, val_loader)
            
            # 保存最佳模型
            if val_accuracy > best_accuracy:
                best_accuracy = val_accuracy
                torch.save(
                    model.state_dict(),
                    f'best_model_epoch_{epoch}.pth'
                )
            
            # 更新学习率
            scheduler.step()
            
            # 打印进度
            print(f'Epoch {epoch+1}/{self.config["epochs"]}: '
                  f'Train Loss: {train_loss:.4f}, '
                  f'Val Accuracy: {val_accuracy:.4f}')
        
        return model
    
    def train_one_epoch(self, model, train_loader, optimizer, criterion):
        """训练一个 epoch"""
        model.train()
        total_loss = 0
        
        for images, labels in train_loader:
            images, labels = images.to(self.device), labels.to(self.device)
            
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, labels)
            loss.backward()
            
            # 梯度裁剪
            torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
            
            optimizer.step()
            total_loss += loss.item()
        
        return total_loss / len(train_loader)
    
    def validate(self, model, val_loader):
        """验证"""
        model.eval()
        correct = 0
        total = 0
        
        with torch.no_grad():
            for images, labels in val_loader:
                images, labels = images.to(self.device), labels.to(self.device)
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        accuracy = correct / total
        return accuracy

# 配置
config = {
    'data_root': 'data/corneal_images',
    'batch_size': 16,
    'epochs': 50,
    'num_classes': 2
}

# 训练
pipeline = TrainingPipeline(config)
model = pipeline.train()
```
